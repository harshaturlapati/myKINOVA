269 void setCMD() {
270
271 dynamics->setPosition(q);
272 dynamics->setVelocity(qd);
273 dynamics->setAcceleration(qdd);
274 dynamics->inverseDynamics();
275
276 for (i = 0; i < ACTUATOR_COUNT; ++i)
277 {
278 if (i >= focus)
279 {
280
281 if (CTRL_MODE == 0)
282 {
283 ext_tau[i] = (K[i] * calculate_delta_q(des_q[i], q[i], 'r'));
284 }
285 else if (CTRL_MODE == 1)
286 {
287 ext_tau[i] = (K[i] * calculate_delta_q(des_q[i], q[i], 'r')) + tau_ctrl[i];
288 }
289 else if (CTRL_MODE == 2)
290 {
291 ext_tau[i] = tau_ctrl[i];
292 }
293
294 if (ext_tau[i] > ext_tau_limit[i])
295 {
296 ext_tau[i] = ext_tau_limit[i];
297 }
298
299 if (ext_tau[i] < -ext_tau_limit[i])
300 {
301 ext_tau[i] = -ext_tau_limit[i];
302 }
303
304 tau_cmd[i] = dynamics->getTorque()[i] + (B[i] * qd[i]) + ext_tau[i];
305 base_command.mutable_actuators(i)->set_torque_joint(tau_cmd[i]);
306 base_command.mutable_actuators(i)->set_position(base_feedback.actuators(i).position());

489 float calculate_delta_q(float q_m, float q_s, char unit) {
490 float del_q = q_m - q_s;
491 if (unit == 'd')
492 {
493 if (abs(del_q) <= 180.0)
494 {
495 return del_q;
496 }
497 else if (q_m < 180.0)
498 {
499 return del_q + 360.0;
500 }
501 else
502 {
503 return del_q - 360.0;
504 }
505 }
506 else if (unit == 'r')
507 {
508 if (abs(del_q) <= M_PI)
509 {
510 return del_q;
511 }
512 else if (q_m < M_PI)
513 {
514 return del_q + (2 * M_PI);
515 }
516 else
517 {
518 return del_q - (2 * M_PI);
519 }
520 }

523 bool goHOME(k_api::Base::BaseClient* base_in)
524 {
525 // Make sure the arm is in Single Level Servoing before executing an Action
526 auto servoingMode = k_api::Base::ServoingModeInformation();
527 servoingMode.set_servoing_mode(k_api::Base::ServoingMode::SINGLE_LEVEL_SERVOING);
528 base_in->SetServoingMode(servoingMode);
529 std::this_thread::sleep_for(std::chrono::milliseconds(500));
530
531 // Move arm to ready position
532 std::cout << "Moving the arm to a safe position" << std::endl;
533 auto action_type = k_api::Base::RequestedActionType();
534 action_type.set_action_type(k_api::Base::REACH_JOINT_ANGLES);
535 auto action_list = base_in->ReadAllActions(action_type);
536 auto action_handle = k_api::Base::ActionHandle();
537 action_handle.set_identifier(0);
538 for (k_api::Base::Action action : action_list.action_list())
539 {
540 if (action.name() == "Home")
541 {
542 action_handle = action.handle();
543 }
544 }
545
546 if (action_handle.identifier() == 0)
547 {
548 std::cout << "Can't reach safe position, exiting" << std::endl;
549 return false;
550 }
551 else
552 {
553 // Connect to notification action topic
554 std::promise<k_api::Base::ActionEvent> promise;
555 auto future = promise.get_future();
556 const auto notification_handle = base_in->OnNotificationActionTopic(
557 create_action_event_listener_by_promise(promise),
558 k_api::Common::NotificationOptions{}
559 );
560
561 base_in->ExecuteActionFromReference(action_handle);
562
563 // Wait for action to finish
564 const auto status = future.wait_for(TIMEOUT_DURATION);
565 base_in->Unsubscribe(notification_handle);
566
567 if (status != std::future_status::ready)
568 {
569 std::cout << "Timeout on action notification wait" << std::endl;
570 return false;
571 }
572
573 return true;
574
575 }

578 std::function<void(k_api::Base::ActionNotification)> create_action_event_listener_by_promise(std::promise<k_api::Base::ActionEvent>& finish_promise)
579 {
580 return [&finish_promise](k_api::Base::ActionNotification notification)
581 {
582 const auto action_event = notification.action_event();
583 switch (action_event)
584 {
585 case k_api::Base::ActionEvent::ACTION_END:
586 case k_api::Base::ActionEvent::ACTION_ABORT:
587 finish_promise.set_value(action_event);
588 break;
589 default:
590 break;
591 }
