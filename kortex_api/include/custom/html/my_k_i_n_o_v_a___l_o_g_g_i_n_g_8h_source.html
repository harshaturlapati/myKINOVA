319 void replace_all(
320 std::string& s,
321 std::string const& toReplace,
322 std::string const& replaceWith
323 ) {
324 std::string buf;
325 std::size_t pos = 0;
326 std::size_t prevPos;
327
328 // Reserves rough estimate of final size of string.
329 buf.reserve(s.size());
330
331 while (true) {
332 prevPos = pos;
333 pos = s.find(toReplace, pos);
334 if (pos == std::string::npos)
335 break;
336 buf.append(s, prevPos, pos - prevPos);
337 buf += replaceWith;
338 pos += toReplace.size();
339 }
340
341 buf.append(s, prevPos, s.size() - prevPos);
342 s.swap(buf);
343 }
