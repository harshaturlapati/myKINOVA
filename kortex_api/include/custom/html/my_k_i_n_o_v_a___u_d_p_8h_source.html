98 int initialize_Winsock()
99 {
100 //----------------------
101 // Initialize Winsock
102 iResult = WSAStartup(MAKEWORD(2, 2), &wsaData);
103 if (iResult != NO_ERROR) {
104 printf("WSAStartup failed: %d\n", iResult);
105 return 1;
106 }
107
108 //----------------------
109 // Create a SOCKET for connecting to server - SEND
110 ConnectSocket = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
111 if (ConnectSocket == INVALID_SOCKET) {
112 printf("Error at socket(): %ld\n", WSAGetLastError());
113 WSACleanup();
114 return 1;
115 }
116
117 // Create a receiver socket to receive datagrams - RECV
118 RecvSocket = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
119
120 // Make port non-blocking - VERY IMPORTANT
121 unsigned long ul = 1;
122 int nRet;
123 nRet = ioctlsocket(RecvSocket, FIONBIO, (unsigned long*)&ul);
124
125 iVal = 1; // if you set this too low, the recvsocket will be impatient and might not
126 ret = setsockopt(RecvSocket, SOL_SOCKET, SO_RCVTIMEO, (char*)&iVal, sz);
127
128 if (RecvSocket == INVALID_SOCKET) {
129 wprintf(L"socket failed with error %d\n", WSAGetLastError());
130 return 1;
131 }
132
133 //----------------------
134 // The sockaddr_in structure specifies the address family,
135 // IP address, and port of the server to be connected to. - SEND
136 clientService.sin_family = AF_INET;
137 clientService.sin_addr.s_addr = inet_addr(SEND_IP_ADDRESS);
138 clientService.sin_port = htons(SEND_PORT);
139
140 // Bind the socket to any address and the specified port - RECV
141 RecvAddr.sin_family = AF_INET;
142 RecvAddr.sin_addr.s_addr = inet_addr(RECV_IP_ADDRESS);
143 RecvAddr.sin_port = htons(RECV_PORT);
144
145 iResult2 = ::bind(RecvSocket, (SOCKADDR*)&RecvAddr, sizeof(RecvAddr));
146 if (iResult2 != 0) {
147 wprintf(L"bind failed with error %d\n", WSAGetLastError());
148 return 1;
149 }
150
151 printf("Connect instead of bind was run.\n");
152 if (iResult2 == -1) {
153 printf("Connect instead of bind failed.\n");
154 exit(2);
155 }
156
157 // Connect to server.
158 iResult = connect(ConnectSocket, (SOCKADDR*)&clientService, sizeof(clientService));
159 if (iResult == SOCKET_ERROR) {
160 closesocket(ConnectSocket);
161 printf("Unable to connect to server: %ld\n", WSAGetLastError());
162 WSACleanup();
163 return 1;
164 }
165
166 // Send an initial buffer
167 iResult = send(ConnectSocket, sendbuf, (int)strlen(sendbuf), 0);
168 if (iResult == SOCKET_ERROR) {
169 printf("send failed: %d\n", WSAGetLastError());
170 closesocket(ConnectSocket);
171 WSACleanup();
172 return 1;
173 }
